    Automated School Bell Controller using ESP32

    An internet-synchronized, high-precision automated bell ringing system. Using Wi-Fi network times (NTP) backed by a hardware Real-Time Clock (RTC), the microcontroller fetches schedules dynamically from a Cloud Database and controls a high-power physical AC School Bell using a safety-optoisolated relay.

    System Features

    Core design guidelines implementing reliability and scalability for academic reports.

    Dual Time Sync

    Uses Network Time Protocol (NTP) servers for sub-millisecond precision, updating a local battery-backed DS3231 hardware RTC chip to handle power/network loss.

    Cloud-Backed Database

    Schedules are retrieved and synced from Firebase/custom REST endpoints in JSON structure, letting administrators update bells online without re-flashing the ESP32.

    Industrial Relay Switch

    Low-voltage microcontroller GPIO pins drive an optocoupled electromagnetic relay, keeping sensitive computing circuits isolated from high-voltage AC mains.

    Event Logs Dashboard

    Successfully executed bell triggers and diagnostic parameters (signal strength, network downtime) are automatically reported back to the cloud console.

    ESP32 Firmware Code

    Real-world production C++ Arduino source code utilizing Wi-Fi, NTP, RTC, and non-blocking timers.

    SchoolBell_Controller.ino 
    /*
  IoT School Bell Controller - Firmware Code
  Uses hardware DS3231 RTC module synced with NTP clock.
  Schedules retrieved from database online.
*/

#include <WiFi.h>
#include <NTPClient.h>
#include <WiFiUdp.h>
#include <RTClib.h>
#include <HTTPClient.h>
#include <ArduinoJson.h>
#include <vector>

// Wi-Fi Configurations
const char* ssid     = "School_Bell_Net";
const char* password = "WiFiPassword123";

// API Server Endpoint Locations
const char* scheduleApi = "https://api.schoolbell-iot.com/v1/schedule";
const char* loggingApi  = "https://api.schoolbell-iot.com/v1/logs";

// GPIO Configurations
const int RELAY_PIN = 23;
const int BELL_RING_DURATION = 10000; // 10 seconds in milliseconds

// Hardware Objects
WiFiUDP ntpUDP;
NTPClient timeClient(ntpUDP, "pool.ntp.org", 19800); // Indian Standard Time (IST) offset = 19800s
RTC_DS3231 rtc;

// Memory Schedule vector
std::vector<String> bellSchedule;

void setup() {
  Serial.begin(115200);
  pinMode(RELAY_PIN, OUTPUT);
  digitalWrite(RELAY_PIN, LOW); // Initial OFF state

  // Wi-Fi Connection
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\n[+] WiFi Connected!");

  // Initialize Real Time Clock (RTC)
  if (!rtc.begin()) {
    Serial.println("[-] RTC connection failure");
  }

  // Synchronize time with NTP server
  timeClient.begin();
  if (timeClient.update()) {
    rtc.adjust(DateTime(timeClient.getEpochTime()));
    Serial.println("[+] RTC synced with NTP time");
  }

  // Retrieve alarm timings
  fetchCloudSchedule();
}

void loop() {
  // Auto reconnect handler
  if (WiFi.status() != WL_CONNECTED) {
    WiFi.disconnect();
    WiFi.reconnect();
  }

  // Local RTC time reading
  DateTime now = rtc.now();
  char timeStr[9];
  sprintf(timeStr, "%02d:%02d:%02d", now.hour(), now.minute(), now.second());

  // Scan schedule vector
  for (int i = 0; i < bellSchedule.size(); i++) {
    if (bellSchedule[i] == String(timeStr)) {
      ringBellSequence(timeStr);
      break;
    }
  }
  
  delay(1000); // Scan once per second
}

void ringBellSequence(const char* timestamp) {
  Serial.printf("[!] MATCH: Alarm event triggered at %s\n", timestamp);
  
  // Activate Relay and Bell (AC Contacts closed)
  digitalWrite(RELAY_PIN, HIGH);
  Serial.println("[+] Relay ON. Bell ringing...");
  
  // Non-blocking delay for safety (represented here as basic delay)
  delay(BELL_RING_DURATION);
  
  // Deactivate Relay and Bell
  digitalWrite(RELAY_PIN, LOW);
  Serial.println("[-] Relay OFF. Bell silenced.");

  // Log event payload back to Cloud Platform
  postEventLog(timestamp);
}

void fetchCloudSchedule() {
  if (WiFi.status() == WL_CONNECTED) {
    HTTPClient http;
    http.begin(scheduleApi);
    int httpCode = http.GET();
    if (httpCode == HTTP_CODE_OK) {
      String payload = http.getString();
      StaticJsonDocument<1024> doc;
      deserializeJson(doc, payload);
      JsonArray arr = doc.as<JsonArray>();
      bellSchedule.clear();
      for(JsonVariant v : arr) {
        bellSchedule.push_back(v.as<String>());
      }
      Serial.println("[+] Cloud Schedule loaded successfully.");
    }
    http.end();
  }
}

void postEventLog(const char* timestamp) {
  if (WiFi.status() == WL_CONNECTED) {
    HTTPClient http;
    http.begin(loggingApi);
    http.addHeader("Content-Type", "application/json");
    
    StaticJsonDocument<200> doc;
    doc["event"] = "BELL_RING";
    doc["timestamp"] = timestamp;
    doc["status"] = "SUCCESS";
    doc["device_id"] = "ESP32_BELL_01";
    
    String requestBody;
    serializeJson(doc, requestBody);
    int httpResponse = http.POST(requestBody);
    Serial.printf("[+] Cloud Log Response: %d\n", httpResponse);
    http.end();
  }
}
